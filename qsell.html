# Product Model with Boosting Logic
class Product(models.Model):
    title = models.CharField(max_length=200)
    is_boosted = models.BooleanField(default=False)
    boost_start = models.DateTimeField(null=True)
    boost_expiry = models.DateTimeField(null=True)

    def activate_boost(self):
        now = timezone.now()
        self.is_boosted = True
        self.boost_expiry = now + timedelta(days=30)
        self.save()

    @property
    def is_boost_active(self):
        return self.is_boosted and self.boost_expiry > timezone.now()
